        }

// делаем ункцию Make2raza которая двыажды(или трижды) возносит число в квадрат
int Make2raza (
    Func<int, int> f, // f - функция, параметр функции Make2raza (тут мы не задаем ей параметр мы эт потом делаем)
    int a) 
// Func - тип данных, обозначает функцию (как string или int и тд)
// в угловых скобках сначала параметр (че принимает), а потом результат (че вовращает) (мб наоборот)  х - название параметра
{
    int pervyraz = f(a);  
    int vtoroyraz = f(pervyraz);
    return vtoroyraz;

}

    Console.WriteLine (Make2raza (f:Count,5));  // Функцию Count мы не вызываем, а используем как значение параментар то есть скобок после нее не надо (в последующих вариках мы вместо того чтобы использовать эту функцию создаем новую одноразовую)
    Console.WriteLine (Make2raza (f:(n => n*n),6));  // Когда мы дходим до этого момента а принимает значение 6, потом возвращается обратно к int pervyraz = f(a);  int vtoroyraz = f(pervyraz); ,  где показывается как выполняется функция f ну и выполняется
    Console.WriteLine (Make2raza (f:(k => k*k*k),2)); 
    //f - одноразовая функция (мы ее вместо Count юзаем). Ей не надо быть дох расписаной, так что расписываем ее прям сдесь. n - ее параметр, а через "=>" мы пишем  n*n , то есть тело. Ну и эту же функцию как видно мы юзаем выше, пока расписываем тело Make2raza
// насчет чисел 6, 2 в этих последних вдух строчках. в параметрах при расписывании Make2raza мы указали некую переменную int a. и в строчке "int pervyraz = f(a);  " мы указали а как параметр одноразовой функции f. и теперь уже при расписывании функции f мы обозначаем значение этой переменной в конце (также оно является значением в первом случае n , во втором случае k (чтобы расписать функцию мы юзаем переменные (n, k)))
